function parseKml(kml) { const poles = { te: [] }; // Presuming you have a structure for poles.te const folders = kml.getElementsByTagName('Folder'); for (let folder of folders) { const folderName = folder.getElementsByTagName('name')[0].textContent; // Add T7, T9, and TL as TE categories if (['T7', 'T9', 'TL'].includes(folderName)) { poles.te.push(folderName); } } // Further parsing logic... return poles; }
